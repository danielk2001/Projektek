#adat file beolvasása
fraud <- read.csv("~/Desktop/fraudTrain.csv")
#Metrikus MDS
mfraud <- fraud[fraud$is_fraud==1,] #Adatok szűrése a csalásokra

fraud_mds <- data.frame(mfraud[, c(6,12)])

#500 elemű véletlen minta vételezés 
set.seed(43)
mfraud_mds <- fraud_mds[sample(1:nrow(fraud_mds), size = 500, replace = FALSE), ]

sumbygroup <- aggregate(mfraud_mds$amt ~ mfraud_mds$state, data = mfraud_mds, sum) #amt változó államonként csoportosított szummázása
state <- data.frame(sumbygroup[,1])
sumbygroup <- sumbygroup[,-1]
sumbygroup <- data.frame(scale(sumbygroup)) #standardizálás

tav <- as.matrix(dist(sumbygroup)) #tav mártix létrehozása
mds_metric<- mds(tav, ndim = 2) # metrikus skálázás
mds_metric$stress 
## [1] 5.988139e-15
plot(mds_metric$conf)
text(mds_metric$conf, labels=state$sumbygroup...1.)

#Nem-Metrikus-MDS
mds_nonmetric<- mds(tav, ndim = 2, type="ordinal") #nem metrikus (ordinális) skálázás
mds_nonmetric$stress
## [1] 3.630195e-16
plot(mds_nonmetric$conf)
text(mds_nonmetric$conf, labels=state$sumbygroup...1.)
